# Yuelin's Compiler

.data
	global:
	
	.space 4
	STR_1: .asciiz "j = "
	STR_2: .asciiz "\n"

.text
	la $gp, global
	j FUNC_main
FUNC_main:
	# DEF, INT(j[sp-0x4]), 12
	li $5, 12
	# PRINT_STR STR_1
	li $2, 4
	la $4, STR_1
	syscall
	# PRINT_INT INT(j[sp-0x4])
	li $2, 1
	move $4, $5
	syscall
	# PRINT_STR STR_2
	li $2, 4
	la $4, STR_2
	syscall
	# RETURN 0
	li $2, 10
	syscall
