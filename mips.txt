# Yuelin's Compiler
	
.data
	global:
	
	.space 4
	STR_1: .asciiz "b = "
	STR_2: .asciiz "\n"
.text
	la $gp, global
	j FUNC_main
FUNC_main:
	# JUMP INLINE_BEGIN_1
INLINE_BEGIN_1:
	# PRINT_STR STR_1
	li $2, 4
	la $4, STR_1
	syscall
	# PRINT_INT INT(b[sp-0x4])
	li $2, 1
	move $4, $5
	syscall
	# PRINT_STR STR_2
	li $2, 4
	la $4, STR_2
	syscall
	# JUMP INLINE_END_1
INLINE_END_1:
	# RETURN 0
	li $2, 10
	syscall
