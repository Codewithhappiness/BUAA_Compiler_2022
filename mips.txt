# Yuelin's Compiler
	
.data
	global:
	
	.space 4
	STR_1: .asciiz "19241091\n"
	STR_2: .asciiz "==================testfile4 end=================="
.text
	la $gp, global
	j FUNC_main
FUNC_main:
	# PRINT_STR STR_1
	li $2, 4
	la $4, STR_1
	syscall
	# PRINT_STR STR_2
	li $2, 4
	la $4, STR_2
	syscall
	# RETURN 0
	li $2, 10
	syscall
